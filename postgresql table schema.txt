CREATE TABLE channels (id SERIAL PRIMARY KEY, name varchar(80), background varchar(150));
CREATE TABLE videos (id SERIAL PRIMARY KEY, url varchar(20), channel_id integer);
CREATE TABLE users (id SERIAL PRIMARY KEY, name varchar(80));
CREATE TABLE likes (id SERIAL PRIMARY KEY, start_time integer, stop_time integer, video_id integer, channel_id integer);
CREATE TABLE likes_by_user (user_id integer, likes_id integer);
CREATE TABLE ignores (user_id integer, video_id integer);
